function sumOfSquares(n) {
  if (typeof _ !== 'undefined') {
    return _.chain(n).times(x => x * x).sum().value();
  } else {
    return Array.from({ length: n }, (v, i) => i * i).reduce((a, b) => a + b);
  }
}

  }
}
